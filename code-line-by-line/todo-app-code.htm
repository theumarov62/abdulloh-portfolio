ui-changer.js

export function uiChanger(todos) {
  elTodosContainer.innerHTML = "";
  todos.forEach((element) => {
    const clone = elCardTemplate.content.cloneNode(true);
    clone.querySelector("h5").innerText = element.title;
    clone.querySelector("p").innerText = element.description;
    elTodosContainer.append(clone);
  });
}
